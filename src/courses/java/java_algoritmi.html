Primer: Upotreba algoritama nad LinkedList kolekcijom
import java.util.*; public class AlgoritmiZaRadSaKolekcijama { public static void main(String[] args) { //Pravi i inicijalizuje povezanu listu LinkedList<Integer>ll = new LinkedList<Integer>(); ll.add(-8); ll.add(20); ll.add(-20); ll.add(8); // Ispis sadrzaja lsite System.out.print("Sadrzaj liste: "); for (int i:ll) System.out.print(i + " "); System.out.println(); // Sortiranje liste Collections.sort(ll); System.out.print("Sortirana lista:: "); for (int i:ll) System.out.print(i + " "); System.out.println(); // Prinalazenje pozicije elementa Collections.binarySearch(ll,8); System.out.print("8 je na poziciji " + Collections.binarySearch(ll,8) + " u listi"); System.out.println(); // Nasumicno mesanje sadrzaja liste Collections.shuffle(ll); // Prikazivanje sadrzaja liste System.out.print("Sadrzaj liste nakon mesanja: "); for (int i:ll) System.out.print(i + " "); System.out.println(); // Pronalazenje najvece i najmanje vrednosti System.out.println(); System.out.println("Najmanja vrednost: " + Collections.min(ll)); System.out.println("Najveca vrednost: " + Collections.max(ll)) } }
Rezultat programa:
Sadrzaj liste: -8 20 -20 8 Sortirana lista:: -20 -8 8 20 8 je na poziciji 2 u listi Sadrzaj liste nakon mesanja: 8 -8 20 -20 Najmanja vrednost: -20 Najveca vrednost: 20
